```python
def insert(self, n):
    self.heap.append(n)
    i = self.size()
```

```python
def delete(self):
    parent = 1
    child = 2
```

### 상향식 힙 만들기

```python
class BHeap:
    def __init__(self, a):
        self.a = a
        self.N = len(a) - 1
    def create_heap(self):
        for i in range(self.N // 2, 0, -1):
            self.downheap(i)

    def insert(self, n):
        self.a.append(n)
        self.N += 1
        self.upheap(self.N)

    def delete_min(self):
        if self.N == 0:
            return None
        minimum = self.a[1]
        self.a[1], self.a[-1] = self.a[-1], self.a[1]
        del self.a[-1]
        self.N -= 1
        self.downheap(1)
        return minimum

    def upheap(self, n):
        while n > 1 and self.a[n // 2][0] > self.a[n][0]:
            self.a[n], self.a[n // 2] = self.a[n // 2], self.a[n]
            n = n // 2

    def dowmheap(self, n):
        while n < self.N // 2:
            k = n * 2
            if k < self.N and self.a[k][0] < self.a[k+1][0]:
                k = k+1
            if self.a[n][0] < self.a[k][0]:
                break
            self.a[n], self.a[k] = self.a[k], self.a[n]
            n = k
```

삽입, 삭제: O(logN)

## 이진 탐색 트리 (15~20)

1. 이진 탐색 트리
  - key 값을 10개정도 쭉 주고 나서 이진 탐색트리로 구현하라
  - 이진 탐색 트리를 주고 나서, 노드 삭제 후 최종 이진 탐색 트리 기술하라


```python
class Node:
    def __init__(self, key, value, left=None, right=None):
        self.key = key
        self.value = value
        self.left = left
        self.right = right

class BST:
    def __init__(self):
        self.root = None

    def get(self, key):
        return self.get_item(self.root, k)

    def get_item(self, n, k):
        if n == None:
            return None
        if n.key > k:
            return get_item(n.left, k)
        if n.key < k:
            return get_item(n.right, k)
        return n.value

    def put(self, key, value):
        self.root = self.put_item(self.root, key, value)

    def put_item(self, n, k, val):
        if n == None:
            return None
        if n.key > k:
            n.left = put_item(n.left, k)
        if n.key < k:
            n.right = put_item(n.right, k)
        n.value = value
        return n

    def min(self):
        if self.root == None:
            return None
        return self.minimum(self.root)

    def minimum(self, n):
        if n.left == None:
            return n
        return self.minimum(n.left)

    def deletemin(self):
        if self.root == None:
            return None
        return self.del_min(self.root)

    def del_min(self, n):
        if n.left == None:
            return n.right
        n.left = self.del_min(n.left)
        return n

    def delete(self, key):
        self.root = self.del_node(self.root, key)

    def del_node(self, n, k):
        if n == None:
            return None
        if n.key > k:
            n.left = del_node(n.left, k)
        elif n.key < k:
            n.right = del_node(n.right, k)
        else:
            if n.right == None:
                return n.left
            if n.left == None:
                return n.right
            target = n
            n = self.minimum(target.right)
            n.right = self.del_min(target.right)
            n.left = target.left
        return n

```

### kruscal

```python
weights = [(0, 1, 9), (0, 2, 10)]
weights.sort(key = lambda t: t[2])
mst = []
N = 7
p = [] * N
def find(a):
    if a != p[a]:
        p[a] = find(p[a])
    return p[a]

def union(u, v):
    root1 = find(u)
    root2 = find(v)
    p[root1] = root2

e = 0
cost = 0
while True:
    if e == N - 1:
        break
    u, v, wt = weights.pop(0)
    if find(u) != find(v):
        union(u, v)
        mst.append((u, v))
        cost += wt
        e += 1
print(cost)
print(mst)
```

O(MlogN)

```python
D[s] = 0

previous[s] = s

for k in range(N):
    m = -1
    min_val = sys.maxsize
    for j in range(N):
        if not visited[j] and D[j] < min_val:
            m = j
            min_val = D[j]
    visited[m] = True
    for v, wt in list(g[m]):
        if not visited[v]:
            if wt < D[v]:
                D[v] = wt
                previous[v] = m

```
