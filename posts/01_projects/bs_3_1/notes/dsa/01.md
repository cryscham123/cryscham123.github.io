## Single Linked List

```python
class SList:
    class Node:
        def __init__(self, item, link):
            self.item = item
            self.link = link

    def __init__(self):
        self.head = None
        self.size = 0

    def is_empty(self):
        return self.size == 0
    
    def size(self):
        return self.size
    
    def insert_front(self, item):
        if self.is_empty():
            self.head = Node(item, None)
        else:
            self.head = Node(item, self.head)
        self.size += 1

    def insert_after(self, item, p):
        p.link = Node(item, p.link)
        self.size += 1

    def delete_front(self):
        if self.is_empty():
            return print('error')
        else:
            self.head = self.head.link
        self.size -= 1

    def delete_after(self, p):
        if self.is_empty():
            return print('error')
        p.link = p.link.link
        self.size -= 1

    def search(self. target):
        p = self.head
        for k in range(self.size)
            if p.item == target:
                return k
            p = p.link
        return None

    def print_list(self):
        p = self.head
        while p != None:
            if p.link != None:
                print(p.item, ' -> ', end='')
            else
                print(p.item)
            p = p.link
```
